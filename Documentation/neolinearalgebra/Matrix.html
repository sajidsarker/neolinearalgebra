332        
333        Args:
334            None
335        
336        Returns:
337            float: Determinant of the matrix.
338        '''
339        if self.check_if_square_matrix(self.data) == False:
340            raise ValueError('Matrix determinant cannot be derived as it is a non-square matrix. Please provide a square matrix.')
341
342        matrix, det = self.data.copy(), 0
343        indices = list(range(len(matrix)))
344
345        if len(matrix) == 2 and len(matrix[0]) == 2:
346            return matrix[0][0] * matrix[1][1] - matrix[1][0] * matrix[0][1]

348        for focus_column in indices:
349            principal_minor = matrix[1:].copy()

361        '''Gets the inverse of the matrix.
362        
363        Args:
364            None
365        
366        Returns:
367            object: Matrix object containing the inverse of the matrix.
368        '''
369        if self.check_if_square_matrix(self.data) == False and self.rows < 2:
370            raise ValueError('Matrix is non-invertible as it is a non-square matrix. Please provide a square matrix of dimensions greater than or equal to (2, 2).')
371
372        det = self.determinant()
373
374        if det == 0:
375            raise Exception('Matrix is non-invertible as the determinant is 0.')
376
377        # Calculate the Inverse of the Determinant
378        inv_det = 1 / det
379
380        # Calculate Determinants of Principal Minors
381        adjugate_matrix = []
382        
383        if self.shape == (2, 2):
384            adjugate_matrix.append([self.data[1][1] * inv_det, -self.data[0][1] * inv_det])
385            adjugate_matrix.append([-self.data[1][0] * inv_det, self.data[0][0] * inv_det])
386            return Matrix(adjugate_matrix)
387
388        for i in range(self.rows):
389            adjugate_matrix.append([])
390            for j in range(self.cols):
391                # Retrieve data non-destructively for generating Principal Minors
392                principal_minor = []
393                for row in range(self.rows):
394                    principal_minor.append(self.data[row].copy())
395
396                # Generate Principal Minors
397                for row in range(len(principal_minor)):
398                    principal_minor[row].pop(j)
399                principal_minor.pop(i)
400
401                # Populate Adjugate Matrix
402                det = Matrix(principal_minor).determinant()
403                adjugate_matrix[i].append(det)
404
405        # Calculate the Product of Cofactors, Inverse of Determinant, and Adjugate Matrix
406        for row in range(self.rows):
407            for col in range(self.cols):
408                adjugate_matrix[row][col] *= (-1)**(i + j) * inv_det
409
410        return Matrix(adjugate_matrix).transpose()
